//Pattern

import java.util.Scanner;

public class Program3 {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter N:");
		int n = sc.nextInt();
		if(n%2!=0 && n>4)
			printPattern(n);
		else
			System.out.println("Pattern Not Possible");
		sc.close();
	}
	
	static void printPattern(int n) {
		int leftSlashIndex = 0; // To hold the index of the left diagonal slash
		int rightSlashIndex = n-1; // To hold the index of the right diagonal slash
		for(int i=0; i<n; i++) {
			for(int j=0; j<n; j++) {
				if(j==leftSlashIndex) {
					if(i!=(int)(n/2)) {
						System.out.print("\\ ");
					}
				}
				if(j==rightSlashIndex) {
					System.out.print("/ ");
				}
				if(j!=leftSlashIndex && j!=rightSlashIndex) {
					System.out.print("* ");
				}
			}
			leftSlashIndex++;
			rightSlashIndex--;
			System.out.println();
		}
	}
}
